### 02:00 — 微信支付基础概念（科普 3 分钟）

**[画面：PPT 展示，简洁白色背景]**

> 主持人：

"在写代码之前，我们先把基础概念梳理清楚。微信支付里有几个关键角色：

第一个是**商户号（MCH_ID）**，这是你在微信支付里的身份，类似银行卡号。

第二个是**AppID**，这是你的微信公众号或小程序的唯一标识。

第三个是**API 密钥（APIv3Key）**，用于签名验证，长度 32 位，自己设置，一定要妥善保管。

第四个是**证书（apiclient_cert）**，退款、敏感操作需要用到，今天会用到。"

**[切换到流程图]**

"支付流程分四步：

一、**统一下单**——你的服务器向微信支付 API 发起请求，获得预支付 ID（prepay_id）。

二、**前端拉起支付**——把 prepay_id 和签名参数传给前端，前端调用微信 JS-SDK 或 App SDK 弹出支付界面。

三、**用户支付**——用户在微信里完成付款。

四、**支付回调**——微信异步通知你的服务器，支付成功，你在这里完成订单处理。"

**[停顿，强调]**

"有一个非常重要的原则：**永远不要以前端通知为准，一定要以后端回调为准。** 前端可以伪造，回调是微信签名的。"

### 08:00 — JSAPI 支付实战（5 分钟）

**[画面：代码编辑器全屏]**

> 主持人：

"JSAPI 支付适合在微信公众号网页里使用。用户在公众号里点击购买，直接弹出支付界面，体验非常流畅。

我们先写统一下单接口：

```javascript
// pay/jsapi.js
const WxPay = require('wechatpay-node-v3');
const { readFileSync } = require('fs');

const pay = new WxPay({
  appid: process.env.WX_PAY_APP_ID,
  mchid: process.env.WX_PAY_MCH_ID,
  publicKey: readFileSync('./certs/apiclient_cert.pem'),
  privateKey: readFileSync('./certs/apiclient_key.pem'),
});

// 创建 JSAPI 订单
async function createJsApiOrder(openid, product, amount, orderId) {
  const result = await pay.transactions_jsapi({
    appid: process.env.WX_PAY_APP_ID,
    mchid: process.env.WX_PAY_MCH_ID,
    description: product,
    out_trade_no: orderId,
    notify_url: process.env.WX_PAY_NOTIFY_URL,
    amount: {
      total: Math.round(amount * 100),  // 单位：分
      currency: 'CNY'
    },
    payer: {
      openid: openid  // 用户的微信 openid
    }
  });

  return result.data;  // 包含 prepay_id
}

module.exports = { createJsApiOrder, pay };
```

然后是前端需要的签名参数：

```javascript
// 生成前端支付参数
async function getJsApiParams(prepayId) {
  const timestamp = Math.floor(Date.now() / 1000).toString();
  const nonceStr = Math.random().toString(36).slice(2);
  const packageStr = `prepay_id=${prepayId}`;

  const message = `${process.env.WX_PAY_APP_ID}\n${timestamp}\n${nonceStr}\n${packageStr}\n`;
  const paySign = await pay.sign(message);

  return {
    appId: process.env.WX_PAY_APP_ID,
    timeStamp: timestamp,
    nonceStr,
    package: packageStr,
    signType: 'RSA',
    paySign
  };
}
```

前端调用就很简单了：

```javascript
// 前端代码（在微信公众号 H5 里）
function doPay(params) {
  WeixinJSBridge.invoke('getBrandWCPayRequest', params, (res) => {
    if (res.err_msg === 'get_brand_wcpay_request:ok') {
      // 支付成功 — 但不要在这里做业务逻辑！
      // 只做 UI 提示，等待后端回调
      showSuccess('支付成功！订单处理中...');
    } else {
      showError('支付取消或失败');
    }
  });
}
```

### 13:00 — Native 支付实战（4 分钟）

**[画面：代码编辑器]**

> 主持人：

"Native 支付会给你一个 code_url，你把它转成二维码显示出来，用户扫码付款。适合 PC 端、收银台、大屏幕场景。

代码非常简洁：

```javascript
// pay/native.js
const QRCode = require('qrcode');

async function createNativeOrder(product, amount, orderId) {
  const result = await pay.transactions_native({
    appid: process.env.WX_PAY_APP_ID,
    mchid: process.env.WX_PAY_MCH_ID,
    description: product,
    out_trade_no: orderId,
    notify_url: process.env.WX_PAY_NOTIFY_URL,
    amount: {
      total: Math.round(amount * 100),
      currency: 'CNY'
    }
  });

  const codeUrl = result.data.code_url;

  // 生成 base64 二维码图片
  const qrBase64 = await QRCode.toDataURL(codeUrl, {
    width: 300,
    margin: 2,
    color: {
      dark: '#000000',
      light: '#FFFFFF'
    }
  });

  return {
    orderId,
    codeUrl,
    qrBase64,
    expireAt: Date.now() + 2 * 60 * 60 * 1000  // 2小时有效期
  };
}
```

然后轮询订单状态（或等待回调）：

```javascript
// 轮询查询订单状态（前端每 3 秒查一次）
async function queryOrderStatus(orderId) {
  const result = await pay.query({ out_trade_no: orderId });
  const status = result.data.trade_state;

  // SUCCESS / NOTPAY / CLOSED / REFUND / REVOKED
  return {
    paid: status === 'SUCCESS',
    status,
    amount: result.data.amount?.payer_total / 100  // 转回元
  };
}
```

**[慢速展示代码]**

```javascript
// skills/payment-skill.js

const PRODUCTS = {
  '基础会员': { price: 39, period: '月' },
  '高级会员': { price: 199, period: '年' },
  '企业版': { price: 999, period: '年' },
};

async function paymentSkill(intent, context) {
  const { action, product, openid, clientIp } = intent;

  switch (action) {

    case 'create_order': {
      const productInfo = PRODUCTS[product];
      if (!productInfo) {
        return { success: false, message: `未找到产品：${product}。可选：${Object.keys(PRODUCTS).join('、')}` };
      }

      const orderId = `ORD_${Date.now()}_${Math.random().toString(36).slice(2, 8).toUpperCase()}`;

      // 根据场景自动选择支付方式
      let payResult;
      if (context.platform === 'wechat_mp' && openid) {
        // 公众号 → JSAPI
        const { createJsApiOrder, getJsApiParams } = require('./jsapi');
        const order = await createJsApiOrder(openid, product, productInfo.price, orderId);
        const params = await getJsApiParams(order.prepay_id);
        payResult = { type: 'jsapi', params, orderId };

      } else if (context.platform === 'mobile_browser') {
        // 手机浏览器 → H5
        const { createH5Order } = require('./h5');
        const h5Url = await createH5Order(product, productInfo.price, orderId, clientIp);
        payResult = { type: 'h5', url: h5Url, orderId };

      } else {
        // PC 或其他 → Native 二维码
        const { createNativeOrder } = require('./native');
        const native = await createNativeOrder(product, productInfo.price, orderId);
        payResult = { type: 'native', qr: native.qrBase64, orderId };
      }

      return {
        success: true,
        orderId,
        product,
        amount: productInfo.price,
        payType: payResult.type,
        payData: payResult,
        message: `✅ 订单已创建！\n\n` +
                 `📦 商品：${product}（${productInfo.period}）\n` +
                 `💰 金额：¥${productInfo.price}\n` +
                 `🆔 订单：${orderId}\n\n` +
                 (payResult.type === 'native' ?
                   `请扫码完成支付（有效期 2 小时）` :
                   `请点击链接完成支付`)
      };
    }

    case 'query_order': {
      const { queryOrderStatus } = require('./native');
      const status = await queryOrderStatus(intent.orderId);
      return {
        ...status,
        message: status.paid ?
          `✅ 订单 ${intent.orderId} 已支付成功，金额 ¥${status.amount}` :
          `⏳ 订单 ${intent.orderId} 等待支付中`
      };
    }

    case 'list_products': {
      const list = Object.entries(PRODUCTS)
        .map(([name, info]) => `• ${name}：¥${info.price}/${info.period}`)
        .join('\n');
      return { success: true, message: `📦 可购买商品：\n\n${list}` };
    }

    default:
      return { success: false, message: '未知操作' };
  }
}

module.exports = { paymentSkill };
```

> 主持人：

"最后是回调处理，这是最关键的部分：

```javascript
// pay/notify.js
app.post('/pay/notify', express.text({ type: '*/*' }), async (req, res) => {
  try {
    // 1. 验证微信签名
    const { Verification } = require('wechatpay-node-v3/dist/lib/verification');
    const verified = await Verification.verify(/* 证书 */)(req.headers, req.body);
    if (!verified) return res.status(401).json({ code: 'FAIL', message: '签名验证失败' });

    // 2. 解密通知数据
    const decoded = pay.decipher_gcm(
      req.body.resource.ciphertext,
      req.body.resource.associated_data,
      req.body.resource.nonce,
      process.env.WX_PAY_API_V3_KEY
    );

    const { out_trade_no, trade_state, amount } = JSON.parse(decoded);

    if (trade_state === 'SUCCESS') {
      // 3. 幂等处理——防止重复回调
      const alreadyProcessed = await db.orders.findOne({ orderId: out_trade_no, status: 'paid' });
      if (!alreadyProcessed) {
        await db.orders.update({ orderId: out_trade_no }, { status: 'paid', paidAt: new Date() });
        // 触发业务逻辑：开通会员、发送确认消息...
        await activateMembership(out_trade_no);
        await notifyUser(out_trade_no, amount.payer_total / 100);
      }
    }

    res.json({ code: 'SUCCESS', message: '成功' });
  } catch (err) {
    console.error('回调处理失败:', err);
    res.status(500).json({ code: 'FAIL', message: err.message });
  }
});
```

幂等处理非常重要——微信可能会重复发回调，你要保证同一订单只处理一次。"

## 📝 技术要点速查

| 支付方式 | 适用场景 | 获取什么 | 前端操作 |
|----------|---------|---------|---------|
| JSAPI | 微信公众号网页 | prepay_id + 签名 | 调用 `WeixinJSBridge.invoke` |
| Native | PC/大屏/扫码 | code_url | 转二维码显示 |
| H5 | 手机浏览器 | h5_url | 302 跳转 |

## ⚠️ 安全规范

1. 证书文件（`apiclient_key.pem`）**永远不提交 git**
2. APIv3Key 存在环境变量，不硬编码
3. 回调必须验签，不信任未签名数据
4. 业务逻辑必须在回调中处理，不依赖前端通知
5. 幂等处理：同一订单的回调只执行一次业务逻辑
6. 生产环境必须 HTTPS
