### 账号要求

| 公众号类型 | 消息回复 | 模板消息 | 网页授权 | 自定义菜单 |
|-----------|---------|---------|---------|-----------|
| 订阅号（未认证）| ✅ | ❌ | ❌ | ✅（折叠） |

## 开发环境配置

### 1. 获取公众号凭证

登录 [微信公众平台](https://mp.weixin.qq.com) → 设置与开发 → 基本配置：

```env
# .env 文件
WECHAT_APP_ID=wx1234567890abcdef
WECHAT_APP_SECRET=your_app_secret_here
WECHAT_TOKEN=your_custom_token_here          # 自定义，用于消息验签
WECHAT_ENCODING_AES_KEY=your_aes_key_here    # 消息加密用（可选）
```

### 2. OpenClaw 配置

```yaml
# openclaw.config.yaml
channels:
  wechat-mp:
    type: wechat_mp
    enabled: true
    app_id: ${WECHAT_APP_ID}
    app_secret: ${WECHAT_APP_SECRET}
    token: ${WECHAT_TOKEN}
    encoding_aes_key: ${WECHAT_ENCODING_AES_KEY}  # 启用加密时填写
    # 消息回复模式
    reply_mode: async    # sync（5秒内）/ async（客服消息接口）
    # AI 路由
    ai_router:
      default: gpt-4o
      image_recognition: gpt-4o-vision
      voice_transcription: whisper
```

### 3. 服务器验证配置

在公众平台填写：
- **服务器地址(URL)**：`https://your-domain.com/wechat/callback`
- **令牌(Token)**：与 `.env` 中的 `WECHAT_TOKEN` 一致
- **消息加解密方式**：推荐选「兼容模式」或「安全模式」

## 接入流程

### 完整服务端代码

```javascript
// src/wechat-mp.js
const express = require('express');
const crypto = require('crypto');
const xml2js = require('xml2js');
const { OpenClaw } = require('@openclaw/core');

const app = express();
const claw = new OpenClaw({
  channel: 'wechat-mp',
  aiModel: 'gpt-4o'
});

// 原始 body 解析（微信消息是 XML 格式）
app.use('/wechat', express.raw({ type: 'text/xml' }));
app.use('/wechat', express.text({ type: 'application/xml' }));

// ========== 消息验签工具 ==========
function verifySignature(token, timestamp, nonce, signature) {
  const str = [token, timestamp, nonce].sort().join('');
  const hash = crypto.createHash('sha1').update(str).digest('hex');
  return hash === signature;
}

// ========== 生成 XML 回复 ==========
function buildTextReply(toUser, fromUser, content) {
  const timestamp = Math.floor(Date.now() / 1000);
  return `<xml>
  <ToUserName><![CDATA[${toUser}]]></ToUserName>
  <FromUserName><![CDATA[${fromUser}]]></FromUserName>
  <CreateTime>${timestamp}</CreateTime>
  <MsgType><![CDATA[text]]></MsgType>
  <Content><![CDATA[${content}]]></Content>
</xml>`;
}

// ========== 服务器验证（GET 请求）==========
app.get('/wechat/callback', (req, res) => {
  const { signature, timestamp, nonce, echostr } = req.query;
  const token = process.env.WECHAT_TOKEN;

  if (verifySignature(token, timestamp, nonce, signature)) {
    console.log('✅ 微信服务器验证成功');
    res.send(echostr);
  } else {
    console.error('❌ 签名验证失败');
    res.status(403).send('Forbidden');
  }
});

// ========== 消息接收（POST 请求）==========
app.post('/wechat/callback', async (req, res) => {
  const { signature, timestamp, nonce } = req.query;
  const token = process.env.WECHAT_TOKEN;

  // 验签
  if (!verifySignature(token, timestamp, nonce, signature)) {
    return res.status(403).send('Forbidden');
  }

  // 解析 XML
  let message;
  try {
    const result = await xml2js.parseStringPromise(req.body.toString(), {
      explicitArray: false
    });
    message = result.xml;
  } catch (err) {
    console.error('XML 解析失败:', err);
    return res.send('success');
  }

  const { MsgType, FromUserName, ToUserName } = message;
  console.log(`📨 收到消息 [${MsgType}] from ${FromUserName}`);

  // 处理不同消息类型
  try {
    switch (MsgType) {
      case 'text':
        await handleTextMessage(message, res);
        break;
      case 'voice':
        await handleVoiceMessage(message, res);
        break;
      case 'image':
        await handleImageMessage(message, res);
        break;
      case 'event':
        await handleEventMessage(message, res);
        break;
      default:
        // 不支持的消息类型，返回提示
        res.send(buildTextReply(
          FromUserName,
          ToUserName,
          '暂不支持该消息类型，请发送文字或语音'
        ));
    }
  } catch (err) {
    console.error('处理消息出错:', err);
    res.send(buildTextReply(
      FromUserName,
      ToUserName,
      '系统繁忙，请稍后再试'
    ));
  }
});

// ========== 文本消息处理 ==========
async function handleTextMessage(message, res) {
  const { Content, FromUserName, ToUserName } = message;

  // 关键词快捷响应
  if (Content === '帮助' || Content === 'help') {
    return res.send(buildTextReply(
      FromUserName,
      ToUserName,
      '🤖 OpenClaw 智能助手\n\n' +
      '📌 你可以：\n' +
      '• 直接发消息提问\n' +
      '• 发语音（自动转文字）\n' +
      '• 发图片（AI 识图分析）\n' +
      '• 回复"菜单"查看功能\n\n' +
      '💡 Powered by OpenClaw'
    ));
  }

  // 调用 AI 处理
  // 注意：微信要求 5 秒内回复，超时需用客服消息接口
  const timeoutPromise = new Promise((_, reject) =>
    setTimeout(() => reject(new Error('timeout')), 4000)
  );

  try {
    const aiReply = await Promise.race([
      claw.chat({ userId: FromUserName, content: Content }),
      timeoutPromise
    ]);

    res.send(buildTextReply(FromUserName, ToUserName, aiReply));
  } catch (err) {
    if (err.message === 'timeout') {
      // 超时：先返回 success，再通过客服接口异步发送
      res.send('success');
      // 异步发送（见下方客服消息接口）
      sendCustomerServiceMessage(FromUserName, '正在思考中，稍后回复你...').catch(console.error);
      // 继续等待 AI 结果
      claw.chat({ userId: FromUserName, content: Content })
        .then(reply => sendCustomerServiceMessage(FromUserName, reply))
        .catch(console.error);
    } else {
      throw err;
    }
  }
}

// ========== 语音消息处理 ==========
async function handleVoiceMessage(message, res) {
  const { Recognition, FromUserName, ToUserName } = message;

  // 微信会自动语音识别（服务号开通语音识别后）
  if (Recognition) {
    // 直接使用微信识别结果
    const userText = Recognition;
    const aiReply = await claw.chat({ userId: FromUserName, content: userText });
    res.send(buildTextReply(
      FromUserName,
      ToUserName,
      `🎤 识别内容：${userText}\n\n💬 ${aiReply}`
    ));
  } else {
    res.send(buildTextReply(
      FromUserName,
      ToUserName,
      '请开启语音识别功能，或改为文字输入'
    ));
  }
}

// ========== 图片消息处理 ==========
async function handleImageMessage(message, res) {
  const { PicUrl, FromUserName, ToUserName } = message;

  // 使用视觉模型分析图片
  const analysis = await claw.vision({
    userId: FromUserName,
    imageUrl: PicUrl,
    prompt: '请描述这张图片的内容，如果是文字请帮我识别出来'
  });

  res.send(buildTextReply(FromUserName, ToUserName, `🖼️ 图片分析：\n\n${analysis}`));
}

// ========== 事件消息处理 ==========
async function handleEventMessage(message, res) {
  const { Event, FromUserName, ToUserName, EventKey } = message;

  switch (Event) {
    case 'subscribe':
      // 用户关注
      res.send(buildTextReply(
        FromUserName,
        ToUserName,
        '👋 欢迎关注！\n\n' +
        '我是基于 OpenClaw 驱动的智能助手，随时为你服务。\n\n' +
        '💡 直接发消息来开始对话吧！'
      ));
      // 记录新用户
      await claw.userEvent({ type: 'subscribe', userId: FromUserName });
      break;

    case 'unsubscribe':
      // 用户取消关注
      await claw.userEvent({ type: 'unsubscribe', userId: FromUserName });
      res.send('success');
      break;

    case 'CLICK':
      // 自定义菜单点击
      await handleMenuClick(EventKey, FromUserName, ToUserName, res);
      break;

    case 'SCAN':
      // 扫描带参数二维码
      res.send(buildTextReply(
        FromUserName,
        ToUserName,
        `欢迎通过二维码进入，参数：${EventKey}`
      ));
      break;

    default:
      res.send('success');
  }
}

// ========== 菜单点击处理 ==========
async function handleMenuClick(key, fromUser, toUser, res) {
  const menuActions = {
    'MENU_AI_CHAT': '🤖 AI 对话模式已开启，直接发消息提问吧！',
    'MENU_HELP': '📖 帮助文档：https://docs.openclaw.io',
    'MENU_FEEDBACK': '📮 请发送你的反馈内容，我们会认真阅读！'
  };

  const reply = menuActions[key] || `你点击了菜单：${key}`;
  res.send(buildTextReply(fromUser, toUser, reply));
}

app.listen(3000, () => {
  console.log('🚀 微信公众号服务启动 http://localhost:3000');
});
```

## 消息处理

### 5 秒超时问题解决方案

微信服务器要求 5 秒内响应，超时会重试 3 次。正确处理方式：

```javascript
// 客服消息接口（异步回复）
const axios = require('axios');

// 获取 Access Token（需缓存，有效期 2 小时）
let accessTokenCache = { token: '', expiresAt: 0 };

async function getAccessToken() {
  if (Date.now() < accessTokenCache.expiresAt) {
    return accessTokenCache.token;
  }

  const res = await axios.get('https://api.weixin.qq.com/cgi-bin/token', {
    params: {
      grant_type: 'client_credential',
      appid: process.env.WECHAT_APP_ID,
      secret: process.env.WECHAT_APP_SECRET
    }
  });

  accessTokenCache = {
    token: res.data.access_token,
    expiresAt: Date.now() + (res.data.expires_in - 300) * 1000 // 提前 5 分钟刷新
  };

  return accessTokenCache.token;
}

// 发送客服文本消息
async function sendCustomerServiceMessage(openId, content) {
  const token = await getAccessToken();

  await axios.post(
    `https://api.weixin.qq.com/cgi-bin/message/custom/send?access_token=${token}`,
    {
      touser: openId,
      msgtype: 'text',
      text: { content }
    }
  );
}

// 发送客服图文消息
async function sendNewsMessage(openId, articles) {
  const token = await getAccessToken();

  await axios.post(
    `https://api.weixin.qq.com/cgi-bin/message/custom/send?access_token=${token}`,
    {
      touser: openId,
      msgtype: 'news',
      news: { articles }
    }
  );
}
```

### 发送模板消息

```javascript
// 发送模板消息
async function sendTemplateMessage(openId, templateId, data, url) {
  const token = await getAccessToken();

  const payload = {
    touser: openId,
    template_id: templateId,
    url: url,  // 点击跳转链接
    data: Object.fromEntries(
      Object.entries(data).map(([k, v]) => [k, { value: v, color: '#173177' }])
    )
  };

  const res = await axios.post(
    `https://api.weixin.qq.com/cgi-bin/message/template/send?access_token=${token}`,
    payload
  );

  return res.data;
}

// 使用示例：AI 任务完成后通知用户
async function notifyTaskComplete(openId, taskName, result) {
  await sendTemplateMessage(
    openId,
    'YOUR_TEMPLATE_ID',

      url: 'https://your-domain.com/results'
    },
    'https://your-domain.com/results'
  );
}
```

## 用户管理

### 获取关注用户列表

```javascript
// 获取所有关注者 openId
async function getAllFollowers() {
  const token = await getAccessToken();
  const followers = [];
  let nextOpenId = '';

  do {
    const res = await axios.get(
      `https://api.weixin.qq.com/cgi-bin/user/get?access_token=${token}&next_openid=${nextOpenId}`
    );

    if (res.data.data?.openid) {
      followers.push(...res.data.data.openid);
    }

    nextOpenId = res.data.next_openid || '';
  } while (nextOpenId);

  return followers;
}

// 获取用户信息
async function getUserInfo(openId) {
  const token = await getAccessToken();
  const res = await axios.get(
    `https://api.weixin.qq.com/cgi-bin/user/info?access_token=${token}&openid=${openId}&lang=zh_CN`
  );
  return res.data;
}
```

```javascript
app.get('/auth', (req, res) => {

app.get('/auth/callback', async (req, res) => {
  const { code } = req.query;

  // 通过 code 换取 openId
  const tokenRes = await axios.get('https://api.weixin.qq.com/sns/oauth2/access_token', {
    params: {
      appid: process.env.WECHAT_APP_ID,
      secret: process.env.WECHAT_APP_SECRET,
      code,
      grant_type: 'authorization_code'
    }
  });

  const { openid, access_token } = tokenRes.data;

  // 存储 openId，设置 session 或 JWT
  req.session.openId = openid;

  // 重定向到实际页面
  res.redirect('/app');
});
```

## 高级功能

### 消息去重（避免重试刷重复）

```javascript
const processedMsgIds = new Set();

// 在消息处理开头加入去重逻辑
app.post('/wechat/callback', async (req, res) => {
  const message = await parseXml(req.body);
  const msgId = message.MsgId;

  if (msgId && processedMsgIds.has(msgId)) {
    console.log('重复消息，跳过:', msgId);
    return res.send('success');
  }

  if (msgId) {
    processedMsgIds.add(msgId);
    // 30 分钟后清理（防止内存泄漏）
    setTimeout(() => processedMsgIds.delete(msgId), 30 * 60 * 1000);
  }

  // 继续处理...
});
```

### 多用户会话隔离

```javascript
// 每个用户独立的对话上下文
const userSessions = new Map();

function getUserSession(openId) {
  if (!userSessions.has(openId)) {
    userSessions.set(openId, {
      history: [],
      lastActiveAt: Date.now()
    });
  }
  return userSessions.get(openId);
}

async function chatWithContext(openId, userMessage) {
  const session = getUserSession(openId);
  session.lastActiveAt = Date.now();

  // 保留最近 10 轮对话
  if (session.history.length > 20) {
    session.history = session.history.slice(-20);
  }

  const reply = await claw.chat({
    userId: openId,
    content: userMessage,
    history: session.history
  });

  session.history.push(
    { role: 'user', content: userMessage },
    { role: 'assistant', content: reply }
  );

  return reply;
}
```

## 常见问题

**Q：回复消息总是超时？**
先返回 `success`，再用客服消息接口异步发送结果。参考上方「5 秒超时解决方案」。

**Q：`invalid credential` 错误？**
Access Token 过期（有效期 2 小时）。检查 Token 缓存逻辑，确保过期前刷新。

**Q：能否接入 ChatGPT？**
可以。在 `openclaw.config.yaml` 中配置 `ai_router.default: gpt-4o` 并设置 OpenAI API Key 即可。

**Q：个人主体能否开通服务号？**
个人主体只能开通订阅号。服务号需要企业/个体工商户资质认证。

**Q：消息加密安全模式如何配置？**
设置 `WECHAT_ENCODING_AES_KEY`，并在消息解析前先用 `@wecom/crypto` 解密。
