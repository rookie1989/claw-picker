## 整体架构

```
用户（微信/企业微信）
    │  发送消息："我要买 VIP 会员"
    ▼
OpenClaw Gateway
    │  调用 AI 理解购买意图
    │  提取商品信息
    ▼
支付意图处理模块
    │  调用微信支付 API
    │  生成支付链接 / 二维码
    ▼
用户完成支付
    │
    ▼
微信支付回调（Webhook）
    │  验证签名
    │  更新订单状态
    ▼
OpenClaw 发送确认消息
    "✅ 支付成功！您的 VIP 已开通"
```

## 前置准备

### 1. 微信支付商户号

1. 前往 [pay.weixin.qq.com](https://pay.weixin.qq.com) 申请商户号
2. 完成企业认证（需营业执照）
3. 记录以下信息：
   - **MCHID**：商户号（10 位数字）
   - **APPID**：公众号/小程序/APP 的 AppID
   - **APIv3 密钥**：在「账户中心 → API 安全」中设置

### 2. 证书文件

```bash
# 下载证书（在商户平台「账户中心 → API 安全」中申请）
# 得到以下文件：
# apiclient_cert.p12    — PKCS12 证书
# apiclient_cert.pem    — 证书（公钥）
# apiclient_key.pem     — 私钥

# 将证书放置到安全目录
mkdir -p ~/.openclaw/certs/wxpay
cp apiclient_cert.pem ~/.openclaw/certs/wxpay/
cp apiclient_key.pem  ~/.openclaw/certs/wxpay/
chmod 600 ~/.openclaw/certs/wxpay/*.pem
```

### 3. OpenClaw 配置

```yaml
# ~/.openclaw/config.yaml

plugins:
  wxpay:
    enabled: true
    env: production          # production | sandbox
    
    mchid: "1234567890"      # 商户号
    appid: "wxabcdef123456"  # 公众号/小程序 AppID
    
    # APIv3 密钥（不要放在版本控制中！）
    apiv3_key: "${WXPAY_APIV3_KEY}"
    
    # 证书路径
    cert_path: "~/.openclaw/certs/wxpay/apiclient_cert.pem"
    key_path:  "~/.openclaw/certs/wxpay/apiclient_key.pem"
    
    # 回调地址（必须是 HTTPS）
    notify_url: "https://your-domain.com/webhooks/wxpay"
    
    # 订单超时时间（分钟）
    order_expire_minutes: 30
```

## JSAPI 支付

> 适用于**微信公众号**内的 H5 页面和**微信小程序**。用户在微信内唤起支付键盘，体验最佳。

### 完整流程

```javascript
// plugins/wxpay/jsapi.js

const crypto = require('crypto');
const axios = require('axios');
const fs = require('fs');

class WxPayJSAPI {
  constructor(config) {
    this.mchid = config.mchid;
    this.appid = config.appid;
    this.apiv3Key = config.apiv3_key;
    this.certPath = config.cert_path;
    this.keyPath = config.key_path;
    this.notifyUrl = config.notify_url;
    
    // 加载私钥
    this.privateKey = fs.readFileSync(this.keyPath, 'utf-8');
  }
  
  /**
   * 生成 JSAPI 预支付订单
   * @param {Object} params - 订单参数
   * @returns {Object} - 前端唤起支付所需的参数
   */
  async createOrder(params) {
    const { 
      openid,        // 用户 openid（必须）
      description,   // 商品描述，如 "VIP会员-1个月"
      outTradeNo,    // 商户订单号（唯一）
      totalFee,      // 金额（分）
    } = params;
    
    const body = {
      appid: this.appid,
      mchid: this.mchid,
      description,
      out_trade_no: outTradeNo,
      notify_url: this.notifyUrl,
      amount: {
        total: totalFee,
        currency: 'CNY',
      },
      payer: { openid },
    };
    
    const response = await this._request('POST', 
      '/v3/pay/transactions/jsapi', body);
    
    const { prepay_id } = response.data;
    
    // 生成前端唤起支付所需的签名参数
    return this._buildJSAPIParams(prepay_id);
  }
  
  /**
   * 构建 JSAPI 唤起支付参数
   */
  _buildJSAPIParams(prepayId) {
    const timestamp = Math.floor(Date.now() / 1000).toString();
    const nonceStr = crypto.randomBytes(16).toString('hex');
    const pkg = `prepay_id=${prepayId}`;
    
    const message = [
      this.appid,
      timestamp,
      nonceStr,
      pkg,
    ].join('\n') + '\n';
    
    const paySign = this._sign(message);
    
    return {
      appId: this.appid,
      timeStamp: timestamp,
      nonceStr,
      package: pkg,
      signType: 'RSA',
      paySign,
    };
  }
  
  /**
   * 发起 API 请求（含自动签名）
   */
  async _request(method, path, body) {
    const timestamp = Math.floor(Date.now() / 1000).toString();
    const nonceStr = crypto.randomBytes(16).toString('hex');
    const bodyStr = body ? JSON.stringify(body) : '';
    
    const message = [
      method,
      path,
      timestamp,
      nonceStr,
      bodyStr,
    ].join('\n') + '\n';
    
    const signature = this._sign(message);
    
    // 获取证书序列号（实际项目中从证书解析）
    const serialNo = await this._getCertSerialNo();
    
    const authorization = [
      `WECHATPAY2-SHA256-RSA2048 mchid="${this.mchid}"`,
      `nonce_str="${nonceStr}"`,
      `signature="${signature}"`,
      `timestamp="${timestamp}"`,
      `serial_no="${serialNo}"`,
    ].join(',');
    
    return axios({
      method,
      url: `https://api.mch.weixin.qq.com${path}`,
      data: body,
      headers: {
        'Authorization': authorization,
        'Content-Type': 'application/json',
        'Accept': 'application/json',
      },
    });
  }
  
  /**
   * RSA-SHA256 签名
   */
  _sign(message) {
    return crypto
      .createSign('SHA256')
      .update(message)
      .sign(this.privateKey, 'base64');
  }
  
  /**
   * 读取证书序列号
   */
  async _getCertSerialNo() {
    if (this._serialNo) return this._serialNo;
    const cert = fs.readFileSync(this.certPath, 'utf-8');
    const x509 = new crypto.X509Certificate(cert);
    this._serialNo = x509.serialNumber;
    return this._serialNo;
  }
}

module.exports = WxPayJSAPI;
```

## Native 支付

> 适用于 **PC 网页**或需要展示**二维码**让用户扫码支付的场景。

```javascript
// plugins/wxpay/native.js

class WxPayNative extends WxPayBase {
  /**
   * 创建 Native 支付订单，返回二维码链接
   */
  async createQRCode(params) {
    const { description, outTradeNo, totalFee } = params;
    
    const body = {
      appid: this.appid,
      mchid: this.mchid,
      description,
      out_trade_no: outTradeNo,
      notify_url: this.notifyUrl,
      amount: {
        total: totalFee,
        currency: 'CNY',
      },
    };
    
    const response = await this._request('POST',
      '/v3/pay/transactions/native', body);
    
    const { code_url } = response.data;
    
    // code_url 是 weixin:// 协议的链接
    // 用 qrcode 库生成图片给用户扫描
    return {
      codeUrl: code_url,
      qrImage: await this._generateQRImage(code_url),
    };
  }
  
  async _generateQRImage(codeUrl) {
    const QRCode = require('qrcode');
    return QRCode.toDataURL(codeUrl, {
      width: 300,
      margin: 2,
    });
  }
}
```

## AI 智能收款助手

> 这是本章核心：让 AI 理解用户的购买意图，自动完成下单、出码、发送确认的全流程。

```javascript
// skills/smart-payment/index.js

/**
 * 智能收款 Skill
 * 用户说"买VIP" → AI解析意图 → 创建订单 → 发送支付码
 */
module.exports = {
  name: 'smart-payment',
  version: '1.0.0',
  
  // 触发条件（关键词匹配 + AI 意图识别）
  triggers: [
    { type: 'keyword', patterns: ['购买', '买', '开通', '订阅', '充值', '付款', '支付'] },
    { type: 'intent', model: 'gpt-4o-mini', threshold: 0.85 },
  ],
  
  // 商品目录
  products: {
    'vip_monthly': {
      name: 'VIP 会员（1个月）',
      price: 1900,  // 分，即 ¥19.00
      description: '解锁全部高级功能，有效期 30 天',
    },
    'vip_yearly': {
      name: 'VIP 会员（1年）',
      price: 19800, // ¥198.00
      description: '解锁全部高级功能，有效期 365 天，省 ¥30',
    },
    'credits_100': {
      name: '100 积分包',
      price: 1000,  // ¥10.00
      description: '可用于调用 AI 高级模型，永久有效',
    },
  },
  
  /**
   * 主处理函数
   */
  async handle(message, context, openclaw) {
    const { user, channel } = context;
    
    // Step 1: AI 解析购买意图
    const intent = await openclaw.ai.extract(message.text, {
      schema: {
        product_id: 'string | null',  // 商品 ID
        quantity: 'number',           // 数量
        confirmed: 'boolean',         // 是否明确确认购买
      },
      products: Object.keys(this.products),
      prompt: `
        用户说："${message.text}"
        请判断用户想购买哪个商品（product_id），数量（quantity，默认1），
        以及是否已经明确确认要购买（confirmed，若只是询问则为 false）。
        商品列表：vip_monthly（月度VIP）、vip_yearly（年度VIP）、credits_100（积分包）
      `,
    });
    
    // Step 2: 处理不确定的情况 — 让用户选择
    if (!intent.product_id) {
      return openclaw.reply(message, this._buildProductMenu());
    }
    
    // Step 3: 用户询问但未确认 — 展示详情等待确认
    if (!intent.confirmed) {
      const product = this.products[intent.product_id];
      return openclaw.reply(message, {
        text: `
📦 **${product.name}**
💰 价格：¥${(product.price / 100).toFixed(2)}
📝 说明：${product.description}

确认购买请回复「确认」，或选择其他商品回复商品名称。
        `.trim(),
      });
    }
    
    // Step 4: 创建订单
    const product = this.products[intent.product_id];
    const outTradeNo = `OC${Date.now()}${Math.random().toString(36).slice(2, 7).toUpperCase()}`;
    
    let paymentInfo;
    
    try {
      if (channel.type === 'weixin_mp') {
        // 公众号 JSAPI 支付
        const wxpay = openclaw.plugin('wxpay');
        const order = await wxpay.jsapi.createOrder({
          openid: user.openid,
          description: product.name,
          outTradeNo,
          totalFee: product.price * intent.quantity,
        });
        paymentInfo = { type: 'jsapi', params: order };
        
      } else if (channel.type === 'weixin_work' || channel.type === 'telegram') {
        // 企业微信/Telegram → Native 二维码
        const wxpay = openclaw.plugin('wxpay');
        const { qrImage } = await wxpay.native.createQRCode({
          description: product.name,
          outTradeNo,
          totalFee: product.price * intent.quantity,
        });
        paymentInfo = { type: 'native', qrImage };
      }
    } catch (err) {
      openclaw.logger.error('支付订单创建失败', { err, outTradeNo });
      return openclaw.reply(message, '抱歉，支付系统暂时不可用，请稍后重试或联系客服 🙏');
    }
    
    // Step 5: 保存订单到数据库
    await openclaw.db.orders.create({
      orderId: outTradeNo,
      userId: user.id,
      productId: intent.product_id,
      quantity: intent.quantity,
      totalFee: product.price * intent.quantity,
      status: 'pending',
      createdAt: new Date(),
      expireAt: new Date(Date.now() + 30 * 60 * 1000), // 30 分钟
    });
    
    // Step 6: 发送支付信息给用户
    if (paymentInfo.type === 'jsapi') {
      // JSAPI：发送唤起支付的小程序卡片或模板消息
      return openclaw.reply(message, {
        type: 'pay_card',
        title: `订单确认：${product.name}`,
        amount: `¥${(product.price * intent.quantity / 100).toFixed(2)}`,
        order_id: outTradeNo,
        jsapi_params: paymentInfo.params,
      });
    } else {
      // Native：发送二维码图片
      return openclaw.reply(message, {
        type: 'image_with_text',
        image: paymentInfo.qrImage,
        text: `
🛒 **订单确认**
商品：${product.name}
金额：¥${(product.price * intent.quantity / 100).toFixed(2)}
订单号：${outTradeNo}

请用微信扫描上方二维码完成支付 👆
二维码有效期 **30 分钟**，过期请重新下单。
        `.trim(),
      });
    }
  },
  
  /**
   * 构建商品选择菜单
   */
  _buildProductMenu() {
    const lines = ['请选择您想购买的商品：\n'];
    for (const [id, product] of Object.entries(this.products)) {
      lines.push(`• **${product.name}** — ¥${(product.price / 100).toFixed(2)}`);
      lines.push(`  ${product.description}\n`);
    }
    lines.push('直接回复商品名称即可，例如「月度VIP」');
    return { text: lines.join('\n') };
  },
};
```

## 订单状态追踪与回调

### 配置回调 Webhook

```javascript
// webhooks/wxpay.js

const crypto = require('crypto');

module.exports = async function wxpayWebhook(req, res, openclaw) {
  // Step 1: 验证回调签名
  const { headers, body } = req;
  
  const timestamp = headers['wechatpay-timestamp'];
  const nonce = headers['wechatpay-nonce'];
  const signature = headers['wechatpay-signature'];
  const serial = headers['wechatpay-serial'];
  
  const message = `${timestamp}\n${nonce}\n${JSON.stringify(body)}\n`;
  
  // 获取微信支付平台证书公钥（需定期更新）
  const wxpayPublicKey = await openclaw.plugin('wxpay').getPlatformCert(serial);
  
  const isValid = crypto
    .createVerify('SHA256')
    .update(message)
    .verify(wxpayPublicKey, signature, 'base64');
  
  if (!isValid) {
    openclaw.logger.warn('微信支付回调签名验证失败', { headers });
    return res.status(401).json({ code: 'INVALID_SIGNATURE' });
  }
  
  // Step 2: 解密回调数据
  const { resource } = body;
  const decrypted = openclaw.plugin('wxpay').decryptResource(resource);
  
  const {
    out_trade_no: orderId,
    transaction_id: wxTradeNo,
    trade_state: tradeState,
    amount: { total: totalFee },
    payer: { openid },
  } = decrypted;
  
  // Step 3: 处理支付成功
  if (tradeState === 'SUCCESS') {
    const order = await openclaw.db.orders.findOne({ orderId });
    
    if (!order) {
      openclaw.logger.error('找不到对应订单', { orderId });
      return res.json({ code: 'SUCCESS' }); // 仍然返回成功避免微信重试
    }
    
    if (order.status === 'paid') {
      // 重复回调，直接返回成功
      return res.json({ code: 'SUCCESS' });
    }
    
    // 幂等更新订单状态
    await openclaw.db.orders.update(
      { orderId },
      { 
        status: 'paid', 
        wxTradeNo,
        paidAt: new Date(),
      }
    );
    
    // Step 4: 开通权益
    await openclaw.services.membership.activate({
      userId: order.userId,
      productId: order.productId,
      quantity: order.quantity,
    });
    
    // Step 5: 发送成功通知给用户
    await openclaw.notify(order.userId, {
      channel: order.channelType,
      text: `
✅ **支付成功！**

感谢您的购买 🎉
商品：${order.productName}
金额：¥${(totalFee / 100).toFixed(2)}
订单号：${orderId}

权益已即时生效，立即体验吧！
      `.trim(),
    });
    
    openclaw.logger.info('支付成功处理完成', { orderId, userId: order.userId });
  }
  
  // 必须在 5 秒内响应
  return res.json({ code: 'SUCCESS' });
};
```

```javascript
async function queryOrderStatus(outTradeNo, wxpay) {

  };

  };
```

```javascript
async function refund(params, wxpay) {
  const { outTradeNo, refundAmount, reason } = params;

    refundId: response.data.refund_id,
    outRefundNo,
    status: response.data.status, // PROCESSING | SUCCESS | ABNORMAL
  };
}
```

## 安全最佳实践

### ✅ 必须遵守

1. **API 密钥不入代码仓库** — 使用环境变量或 Secret Manager
   ```bash
   export WXPAY_APIV3_KEY="your-32-char-key"
   ```

2. **验证每一次回调签名** — 跳过验证会导致伪造订单

3. **回调处理幂等** — 微信可能重发回调，要先检查订单是否已处理

4. **HTTPS 强制** — `notify_url` 必须是 HTTPS，且证书合法

5. **金额二次校验** — 收到回调后对比数据库订单金额，防止金额篡改

6. **5 秒内响应回调** — 超时微信会重发，导致重复处理

### ⚠️ 常见陷阱

```javascript
// ❌ 错误：直接信任回调中的金额
if (notify.amount.total > 0) { activateMembership(); }

// ✅ 正确：与数据库中的订单金额比对
const order = await db.orders.findOne({ orderId });
if (notify.amount.total !== order.totalFee) {
  logger.error('金额不匹配！可能存在篡改', { notify, order });
  return; // 不激活，等人工审核
}
```

## 常见问题

**Q: 回调地址必须是公网 HTTPS 吗？**  
A: 是的。本地开发可用 [ngrok](https://ngrok.com) 或 [localtunnel](https://theboroer.github.io/localtunnel-www/) 临时暴露：
```bash
ngrok http 3000
# 得到 https://xxxx.ngrok.io
# 将 https://xxxx.ngrok.io/webhooks/wxpay 填入 notify_url
```

**Q: 支付成功但没收到回调怎么办？**  
A: 微信会在支付后 15 秒、30 秒、1/2/4/8/16/32 分钟共 8 次重试。若仍未收到，用主动查询接口兜底。

**Q: 如何区分沙箱和生产环境？**  
A: 沙箱 API 地址为 `https://api.mch.weixin.qq.com/sandboxnew/`，配置中设置 `env: sandbox` 即可。

**Q: APPID 和商户号必须绑定吗？**  
A: 是的，JSAPI 支付必须在商户平台绑定 APPID 和商户号，否则会报 `APPID_NOT_EXIST` 错误。
