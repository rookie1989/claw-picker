## 一、方案选择

### 架构对比

```
方案 A：直连模式（不推荐）
小程序/H5 ──────────────────→ OpenClaw → AI 模型
           暴露 API Key！❌

方案 B：后端代理模式（推荐）
小程序/H5 → 你的后端服务 → OpenClaw → AI 模型
            隐藏 Key，可鉴权，可限流 ✅

方案 C：云函数模式（推荐·无服务器）
小程序/H5 → 微信云函数 → OpenClaw → AI 模型
            微信生态闭环，无需独立服务器 ✅
```

**推荐选择**：
- 已有后端服务 → 方案 B（后端代理）
- 纯微信生态、想省运维 → 方案 C（云函数）
- 快速原型验证 → 方案 A（仅限内网/测试环境）

  const div = document.createElement('div');
  div.className = `message ${role}`;
  div.innerHTML = `<div class="bubble">${escapeHtml(content)}</div>`;
  container.appendChild(div);
  container.scrollTop = container.scrollHeight;
}

function escapeHtml(text) {
  return text.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');

### 2.2 后端代理服务（Node.js）

```javascript
// server/ai-proxy.js
const express = require('express');
const axios = require('axios');
const rateLimit = require('express-rate-limit');
const router = express.Router();

// 每用户每分钟限 30 次
const limiter = rateLimit({
  windowMs: 60 * 1000,
  max: 30,
  keyGenerator: req => req.user?.id || req.ip
});

router.post('/chat', limiter, authenticate, async (req, res) => {
  const { channel, messages } = req.body;

  // 参数验证
  if (!messages || !Array.isArray(messages)) {
    return res.status(400).json({ error: '参数格式错误' });
  }

  try {
    const response = await axios.post(
      `${process.env.OPENCLAW_URL}/api/v1/chat`,
      { channel: channel || 'h5-assistant', messages },
      {
        headers: { Authorization: `Bearer ${process.env.OPENCLAW_API_KEY}` },
        timeout: 30000
      }
    );
    res.json(response.data);
  } catch (err) {
    console.error('AI proxy error:', err.message);
    res.status(503).json({ error: '服务暂时不可用' });
  }
});

// 身份验证中间件
async function authenticate(req, res, next) {
  const token = req.headers.authorization?.replace('Bearer ', '');
  if (!token) return res.status(401).json({ error: '未登录' });
  
  try {
    req.user = await verifyToken(token); // 你的验证逻辑
    next();
  } catch {
    res.status(401).json({ error: 'Token 无效' });
  }
}

module.exports = router;
```

### 3.2 AI 工具类

```javascript
// utils/ai.js
const BASE_URL = 'https://your-backend.com/api/ai';

// 发送消息并获取 AI 回复
function chat({ channel = 'miniapp-assistant', messages }) {
  return new Promise((resolve, reject) => {
    wx.request({
      url: `${BASE_URL}/chat`,
      method: 'POST',
      header: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${wx.getStorageSync('token') || ''}`
      },
      data: { channel, messages },
      success(res) {
        if (res.statusCode === 200) {
          resolve(res.data.choices[0].message.content);
        } else {
          reject(new Error(res.data?.error || '请求失败'));
        }
      },
      fail(err) {
        reject(new Error(err.errMsg || '网络错误'));
      }
    });
  });
}

// 快捷方法：单次问答
async function ask(question, context = '') {
  const messages = [];
  if (context) messages.push({ role: 'system', content: context });
  messages.push({ role: 'user', content: question });
  return chat({ messages });
}

module.exports = { chat, ask };
```

### 3.4 聊天页面逻辑（JS）

```javascript
// pages/chat/chat.js
const { chat } = require('../../utils/ai');

Page({
  data: {
    messages: [],
    inputText: '',
    loading: false,
    scrollToId: '',
    userAvatar: '',
    quickReplies: ['帮我总结一下', '用简单话解释', '给我举个例子', '还有什么补充？']
  },

  onLoad() {
    // 添加欢迎消息
    this.addMessage('assistant', '你好！我是 AI 助手，有什么可以帮你？');
    
    // 获取用户头像
    const userInfo = wx.getStorageSync('userInfo');
    if (userInfo) this.setData({ userAvatar: userInfo.avatarUrl });
  },

  onInput(e) {
    this.setData({ inputText: e.detail.value });
  },

  async sendMessage() {
    const text = this.data.inputText.trim();
    if (!text || this.data.loading) return;

    this.setData({ inputText: '', loading: true });
    this.addMessage('user', text);

    try {
      // 构建消息历史（最近 10 条）
      const history = this.data.messages.slice(-10).map(m => ({
        role: m.role,
        content: m.content
      }));

      const reply = await chat({
        channel: 'miniapp-assistant',
        messages: history
      });

      this.addMessage('assistant', reply);
    } catch (err) {
      wx.showToast({ title: err.message || '发送失败', icon: 'error' });
    } finally {
      this.setData({ loading: false });
    }
  },

  sendQuickReply(e) {
    this.setData({ inputText: e.currentTarget.dataset.text });
    this.sendMessage();
  },

  addMessage(role, content) {
    const id = Date.now();
    const messages = [...this.data.messages, { id, role, content }];
    this.setData({
      messages,
      scrollToId: 'bottom'
    });
  },

  // 清空对话
  clearChat() {
    wx.showModal({
      title: '清空对话',
      content: '确定要清空当前对话吗？',
      success: (res) => {
        if (res.confirm) {
          this.setData({ messages: [] });
          this.addMessage('assistant', '对话已清空，有什么可以帮你的？');
        }
      }
    });
  }
});
```

### 3.5 使用微信云函数（方案 C）

```javascript
// cloudfunctions/ai-proxy/index.js
const cloud = require('wx-server-sdk');
const axios = require('axios');

cloud.init({ env: cloud.DYNAMIC_CURRENT_ENV });

exports.main = async (event) => {
  const { OPENID } = cloud.getWXContext();
  const { channel, messages } = event;

  if (!OPENID) throw new Error('Unauthorized');

  try {
    const res = await axios.post(
      `${process.env.OPENCLAW_URL}/api/v1/chat`,
      { channel: channel || 'miniapp', messages },
      { headers: { Authorization: `Bearer ${process.env.OPENCLAW_API_KEY}` } }
    );
    return { success: true, data: res.data };
  } catch (err) {
    return { success: false, error: err.message };
  }
};
```

## 四、鉴权与安全

### 4.1 微信登录 + 后端验证

```javascript
// 小程序端：微信一键登录
async function wxLogin() {
  // Step 1: 获取 code
  const { code } = await wx.login();
  
  // Step 2: 发给后端换 Token
  const res = await wx.request({
    url: 'https://your-backend.com/auth/wx-login',
    method: 'POST',
    data: { code }
  });
  
  const { token, userInfo } = res.data;
  wx.setStorageSync('token', token);
  wx.setStorageSync('userInfo', userInfo);
  return token;
}

// 后端：code 换 Token
// POST /auth/wx-login
app.post('/auth/wx-login', async (req, res) => {
  const { code } = req.body;
  
  // 向微信服务器换取 openid + session_key
  const wxRes = await axios.get('https://api.weixin.qq.com/sns/jscode2session', {
    params: {
      appid: process.env.WX_APPID,
      secret: process.env.WX_SECRET,
      js_code: code,
      grant_type: 'authorization_code'
    }
  });
  
  const { openid } = wxRes.data;
  
  // 生成 JWT Token
  const token = jwt.sign({ openid, platform: 'miniapp' }, process.env.JWT_SECRET, { expiresIn: '7d' });
  
  res.json({ token, userInfo: { openid } });
});
```

## 五、流式响应实现

流式响应让用户看到 AI 逐字输出，体验更好：

### H5 流式（SSE）

```javascript
// 前端
async function streamChat(messages) {
  const eventSource = new EventSource(
    `${API_BASE}/stream?messages=${encodeURIComponent(JSON.stringify(messages))}`
  );
  
  let fullText = '';
  eventSource.onmessage = (e) => {
    if (e.data === '[DONE]') {
      eventSource.close();
      return;
    }
    const { delta } = JSON.parse(e.data);
    fullText += delta;
    updateLastMessage(fullText); // 实时更新 UI
  };
}
```

> ⚠️ 微信小程序不支持 EventSource（SSE）。  
> 小程序流式方案：使用 `wx.request` 的 `responseType: 'text'` 配合短轮询，  
> 或直接使用 WebSocket。

### 小程序 WebSocket 流式

```javascript
// 小程序 WebSocket 流式连接
function streamChatWS(messages, onChunk) {
  const ws = wx.connectSocket({
    url: 'wss://your-backend.com/ws/chat'
  });

  ws.onOpen(() => {
    ws.send({
      data: JSON.stringify({
        token: wx.getStorageSync('token'),
        messages
      })
    });
  });

  ws.onMessage((event) => {
    const { type, delta, done } = JSON.parse(event.data);
    if (type === 'delta') onChunk(delta);
    if (done) ws.close();
  });

  return ws;
}

// 页面使用
streamChatWS(history, (delta) => {
  // 追加文字到最后一条消息
  const messages = this.data.messages;
  const last = messages[messages.length - 1];
  last.content += delta;
  this.setData({ messages });
});
```

## 七、常见问题

**Q: 小程序发请求提示「不在以下 request 合法域名列表」？**  
A: 在微信公众平台 → 开发设置 → 服务器域名，添加你的后端域名。开发阶段可在开发者工具中勾选「不校验域名」。

**Q: H5 页面在微信内置浏览器中无法调用 AI？**  
A: 确认后端已配置 CORS，允许微信浏览器的 Referer。

**Q: 如何在小程序中显示 AI 生成的 Markdown？**  
A: 使用 `towxml` 或 `mp-html` 等开源小程序 Markdown 渲染组件。

**Q: 请求超时怎么处理？**  
A: 微信小程序 `wx.request` 默认超时 60s。AI 响应可能较慢，建议：  
- 设置 `timeout: 60000`  
- 展示 Loading 动画减少用户焦虑  
- 使用流式响应提升感知速度
