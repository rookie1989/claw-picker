## 🚀 快速启动监控栈

### docker-compose.monitoring.yml

```yaml
version: '3.8'

services:
  # Prometheus - 指标收集
  prometheus:
    image: prom/prometheus:latest
    restart: always
    ports:
      - "9090:9090"
    volumes:
      - ./monitoring/prometheus:/etc/prometheus
      - prometheus_data:/prometheus
    command:
      - '--config.file=/etc/prometheus/prometheus.yml'
      - '--storage.tsdb.path=/prometheus'
      - '--storage.tsdb.retention.time=30d'
      - '--web.enable-lifecycle'
    networks:
      - monitoring-net

  # Grafana - 可视化
  grafana:
    image: grafana/grafana:latest
    restart: always
    ports:
      - "3030:3000"
    environment:
      - GF_SECURITY_ADMIN_PASSWORD=${GRAFANA_PASSWORD}
      - GF_USERS_ALLOW_SIGN_UP=false
      - GF_SMTP_ENABLED=true
      - GF_SMTP_HOST=smtp.gmail.com:587
    volumes:
      - grafana_data:/var/lib/grafana
      - ./monitoring/grafana/dashboards:/etc/grafana/provisioning/dashboards
      - ./monitoring/grafana/datasources:/etc/grafana/provisioning/datasources
    depends_on:
      - prometheus
    networks:
      - monitoring-net

  # AlertManager - 告警管理
  alertmanager:
    image: prom/alertmanager:latest
    restart: always
    ports:
      - "9093:9093"
    volumes:
      - ./monitoring/alertmanager:/etc/alertmanager
    command:
      - '--config.file=/etc/alertmanager/alertmanager.yml'
    networks:
      - monitoring-net

  # Node Exporter - 系统指标
  node-exporter:
    image: prom/node-exporter:latest
    restart: always
    volumes:
      - /proc:/host/proc:ro
      - /sys:/host/sys:ro
      - /:/rootfs:ro
    command:
      - '--path.procfs=/host/proc'
      - '--path.rootfs=/rootfs'
      - '--path.sysfs=/host/sys'
    networks:
      - monitoring-net

  # Loki - 日志聚合
  loki:
    image: grafana/loki:latest
    restart: always
    ports:
      - "3100:3100"
    volumes:
      - ./monitoring/loki:/etc/loki
      - loki_data:/loki
    command: -config.file=/etc/loki/loki.yml
    networks:
      - monitoring-net

  # Promtail - 日志收集
  promtail:
    image: grafana/promtail:latest
    restart: always
    volumes:
      - /var/log:/var/log:ro
      - ./logs:/openclaw-logs:ro
      - ./monitoring/promtail:/etc/promtail
    command: -config.file=/etc/promtail/promtail.yml
    networks:
      - monitoring-net

volumes:
  prometheus_data:
  grafana_data:
  loki_data:

networks:
  monitoring-net:
    driver: bridge
```

### 启动监控栈

```bash
# 创建监控配置目录
mkdir -p monitoring/{prometheus,grafana/{dashboards,datasources},alertmanager,loki,promtail}

# 启动
docker-compose -f docker-compose.monitoring.yml up -d

# 访问 Grafana
open http://localhost:3030
# 默认账号: admin / your-password
```

### 告警规则

```yaml
# monitoring/prometheus/rules/openclaw.yml

groups:
  - name: openclaw.critical
    interval: 30s
    rules:
      # 节点宕机
      - alert: OpenClawNodeDown
        expr: up{job="openclaw"} == 0
        for: 2m
        labels:
          severity: critical
          channel: pagerduty
        annotations:
          summary: "🔴 OpenClaw 节点宕机"
          description: "节点 {{ $labels.instance }} 已宕机超过 2 分钟"

      # 所有节点不可用
      - alert: OpenClawAllNodesDown
        expr: count(up{job="openclaw"} == 1) == 0
        for: 1m
        labels:
          severity: critical
        annotations:
          summary: "🚨 OpenClaw 服务完全中断"
          description: "所有 OpenClaw 节点均不可用！"

  - name: openclaw.warning
    rules:
      # AI 响应延迟高
      - alert: OpenClawHighLatency
        expr: |
          histogram_quantile(0.95,
            rate(openclaw_ai_request_duration_seconds_bucket[5m])
          ) > 15
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "⚠️ AI 响应延迟过高"
          description: "P95 延迟 {{ $value | humanizeDuration }}，超过 15s 阈值"

      # 错误率高
      - alert: OpenClawHighErrorRate
        expr: |
          rate(openclaw_requests_total{status=~"5.."}[5m]) /
          rate(openclaw_requests_total[5m]) > 0.05
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "⚠️ 请求错误率过高"
          description: "当前错误率 {{ $value | humanizePercentage }}，超过 5% 阈值"

      # 消息队列积压
      - alert: OpenClawQueueBacklog
        expr: openclaw_queue_size > 100
        for: 3m
        labels:
          severity: warning
        annotations:
          summary: "⚠️ 消息队列积压"
          description: "队列中有 {{ $value }} 条待处理消息"

      # AI API 费用过高
      - alert: OpenClawHighAICost
        expr: |
          increase(openclaw_ai_tokens_total[1h]) > 100000
        for: 0m
        labels:
          severity: warning
        annotations:
          summary: "💰 AI Token 用量过高"
          description: "过去 1 小时使用了 {{ $value }} tokens"
```

## 🔔 AlertManager 告警路由

### alertmanager.yml

```yaml
# monitoring/alertmanager/alertmanager.yml

global:
  resolve_timeout: 5m

# 消息模板
templates:
  - '/etc/alertmanager/templates/*.tmpl'

route:
  group_by: ['alertname', 'cluster']
  group_wait: 30s
  group_interval: 5m
  repeat_interval: 12h
  receiver: 'default'
  
  routes:
    # 严重告警 → 企业微信 + 电话
    - match:
        severity: critical
      receiver: wecom-critical
      repeat_interval: 1h
      
    # 警告 → 企业微信群
    - match:
        severity: warning
      receiver: wecom-warning
      repeat_interval: 6h
      
    # 费用告警 → Slack
    - match:
        alertname: OpenClawHighAICost
      receiver: slack-cost
      repeat_interval: 24h

receivers:
  - name: 'default'
    webhook_configs:
      - url: 'http://openclaw:3000/internal/alerts'

  # 企业微信严重告警
  - name: 'wecom-critical'
    webhook_configs:
      - url: 'http://openclaw:3000/internal/notify/wecom'
        send_resolved: true
        http_config:
          bearer_token: "${OPENCLAW_INTERNAL_TOKEN}"

  # 企业微信警告
  - name: 'wecom-warning'
    webhook_configs:
      - url: 'http://openclaw:3000/internal/notify/wecom-group'
        send_resolved: true

  # Slack 费用通知
  - name: 'slack-cost'
    slack_configs:
      - api_url: "${SLACK_WEBHOOK_URL}"
        channel: '#ai-costs'
        title: '💰 AI 费用告警'
        text: '{{ range .Alerts }}{{ .Annotations.description }}{{ end }}'
```

## 📈 Grafana 仪表盘

### 核心指标面板

创建以下仪表盘（JSON 配置见 `monitoring/grafana/dashboards/openclaw.json`）：

**面板 1：请求概览**
```
指标：
- 每秒请求数（RPS）: rate(openclaw_requests_total[1m])
- 成功率: rate(openclaw_requests_total{status="200"}[5m]) / rate(openclaw_requests_total[5m])
- P50/P95/P99 延迟: histogram_quantile(0.95, rate(openclaw_request_duration_seconds_bucket[5m]))
```

**面板 2：AI 性能监控**
```
指标：
- AI 请求延迟分布
- Token 用量趋势
- 按模型分类的使用量
- 估算费用（tokens × 单价）
```

**面板 3：渠道状态**
```
指标：
- 各渠道在线状态
- 各渠道消息量
- 渠道错误率
```

**面板 4：系统资源**
```
指标：
- CPU 使用率
- 内存使用率
- 磁盘 I/O
- 网络流量
```

### Loki 日志聚合配置

```yaml
# monitoring/promtail/promtail.yml

server:
  http_listen_port: 9080

clients:
  - url: http://loki:3100/loki/api/v1/push

scrape_configs:
  - job_name: openclaw-logs
    static_configs:
      - targets:
          - localhost
        labels:
          job: openclaw
          __path__: /openclaw-logs/node*/*.log
    
    pipeline_stages:
      # 解析 JSON 日志
      - json:
          expressions:
            level: level
            message: message
            channel: channel
            user: userId
            
      # 设置时间戳
      - timestamp:
          source: timestamp
          format: RFC3339
          
      # 添加标签
      - labels:
          level:
          channel:
```

### 在 Grafana 中查询日志

```logql
# 查询错误日志
{job="openclaw"} |= "ERROR"

# 查询特定渠道的日志
{job="openclaw", channel="wecom"} | json | level = "error"

# 统计每分钟错误数
rate({job="openclaw"} |= "ERROR" [1m])

# 查询特定用户的所有交互
{job="openclaw"} | json | user = "user123"
```

## 📱 告警通知集成

### OpenClaw 内置告警处理

```javascript
// 在 OpenClaw 中处理 AlertManager Webhook
openclaw.on('internal:alert', async (alert) => {
  const { alertname, severity, annotations, status } = alert;
  
  const emoji = {
    critical: '🔴',
    warning: '⚠️',
    info: 'ℹ️'
  }[severity] || '📢';
  
  const statusEmoji = status === 'resolved' ? '✅' : emoji;
  
  const message = `${statusEmoji} *${alertname}*\n\n${annotations.description}\n\n_${new Date().toLocaleString('zh-CN')}_`;
  
  // 根据严重程度选择通知渠道
  if (severity === 'critical') {
    // 发送到企业微信 + @所有人
    await openclaw.getChannel('wecom-app').sendMessage({
      toparty: 'ops',
      msgtype: 'markdown',
      content: message,
      safe: 0
    });
    
    // 同时发送到 Slack
    await openclaw.getChannel('slack').send({
      channel: '#alerts-critical',
      text: message
    });
  } else {
    // 普通警告只发企业微信群
    await openclaw.getChannel('wecom-webhook').send(message);
  }
});
```

## 🎯 监控最佳实践

### 关键 SLO（服务级别目标）

| 指标 | 目标值 | 告警阈值 |
|------|--------|----------|
| 可用性 | 99.9% | < 99.5% |
| P95 响应时间 | < 10s | > 15s |
| 错误率 | < 1% | > 5% |
| 消息处理延迟 | < 30s | > 60s |

### 告警疲劳防范

```yaml
# 避免告警风暴的配置
route:
  group_wait: 30s        # 等待 30s 汇聚告警
  group_interval: 5m     # 同组告警最少间隔 5 分钟
  repeat_interval: 12h   # 相同告警 12 小时内不重复
  
inhibit_rules:
  # 服务完全宕机时，抑制下级告警
  - source_match:
      alertname: 'OpenClawAllNodesDown'
    target_match_re:
      alertname: 'OpenClawHighLatency|OpenClawHighErrorRate'
    equal: ['cluster']
```
